What is PHPMailer?

PHPMailer is a full-featured email creation and transfer class for PHP. It's the world's most popular code for sending email from PHP, used by many open-source projects including WordPress, Drupal, and Joomla.

Why PHPMailer? It provides a simple and powerful way to send emails with advanced features like SMTP authentication, attachments, HTML emails, and comprehensive error handling.

Installation

Using Composer (Recommended)

The easiest way to install PHPMailer is through Composer:

composer require phpmailer/phpmailer

Manual Installation

You can also download PHPMailer from GitHub and include it manually:

require 'path/to/PHPMailer/src/Exception.php';
require 'path/to/PHPMailer/src/PHPMailer.php';
require 'path/to/PHPMailer/src/SMTP.php';
Note: Make sure your PHP version is 5.5 or higher. PHPMailer 6.x requires PHP 5.5+, while PHPMailer 5.2 supports PHP 5.0+.

Sending Emails with Attachments

PHPMailer makes it incredibly easy to attach files to your emails:

<?php
use PHPMailer\PHPMailer\PHPMailer;

$mail = new PHPMailer(true);

try {
    // SMTP configuration
    $mail->isSMTP();
    $mail->Host = 'smtp.gmail.com';
    $mail->SMTPAuth = true;
    $mail->Username = 'your-email@gmail.com';
    $mail->Password = 'your-app-password';
    $mail->SMTPSecure = 'tls';
    $mail->Port = 587;

    // Email details
    $mail->setFrom('from@example.com', 'Sender');
    $mail->addAddress('recipient@example.com', 'Recipient');
    
    // Add attachments
    $mail->addAttachment('/path/to/document.pdf');
    $mail->addAttachment('/path/to/image.jpg', 'custom-name.jpg');
    
    // Multiple attachments
    $files = ['file1.pdf', 'file2.docx', 'image.png'];
    foreach ($files as $file) {
        $mail->addAttachment($file);
    }
    
    $mail->Subject = 'Email with Attachments';
    $mail->Body = 'Please find the attached files.';
    
    $mail->send();
    echo 'Email sent with attachments!';
} catch (Exception $e) {
    echo "Error: {$mail->ErrorInfo}";
}
?>

Complete Contact Form Example

A production-ready contact form with validation and error handling:

<?php
use PHPMailer\PHPMailer\PHPMailer;
use PHPMailer\PHPMailer\Exception;

require 'vendor/autoload.php';

if ($_SERVER["REQUEST_METHOD"] == "POST") {
    // Sanitize and validate input
    $name = htmlspecialchars(trim($_POST['name']));
    $email = filter_var($_POST['email'], FILTER_VALIDATE_EMAIL);
    $subject = htmlspecialchars(trim($_POST['subject']));
    $message = htmlspecialchars(trim($_POST['message']));
    
    // Validation
    $errors = [];
    
    if (empty($name)) {
        $errors[] = "Name is required";
    }
    
    if (!$email) {
        $errors[] = "Valid email is required";
    }
    
    if (empty($subject)) {
        $errors[] = "Subject is required";
    }
    
    if (empty($message)) {
        $errors[] = "Message is required";
    }
    
    if (empty($errors)) {
        $mail = new PHPMailer(true);
        
        try {
            // Server settings
            $mail->isSMTP();
            $mail->Host = 'smtp.gmail.com';
            $mail->SMTPAuth = true;
            $mail->Username = 'your-email@gmail.com';
            $mail->Password = 'your-app-password';
            $mail->SMTPSecure = PHPMailer::ENCRYPTION_STARTTLS;
            $mail->Port = 587;
            
            // Recipients
            $mail->setFrom($email, $name);
            $mail->addAddress('your-email@gmail.com', 'Your Name');
            $mail->addReplyTo($email, $name);
            
            // Content
            $mail->isHTML(true);
            $mail->Subject = "Contact Form: $subject";
            $mail->Body = "
            <div style='font-family: Arial, sans-serif;'>
                <h2 style='color: #667eea;'>New Contact Form Submission</h2>
                <p><strong>Name:</strong> $name</p>
                <p><strong>Email:</strong> $email</p>
                <p><strong>Subject:</strong> $subject</p>
                <hr>
                <p><strong>Message:</strong></p>
                <p>$message</p>
            </div>
            ";
            
            $mail->AltBody = "Name: $name\nEmail: $email\nSubject: $subject\n\nMessage:\n$message";
            
            $mail->send();
            echo '<div class="success">Thank you! Your message has been sent.</div>';
        } catch (Exception $e) {
            echo "<div class='error'>Error: {$mail->ErrorInfo}</div>";
        }
    } else {
        foreach ($errors as $error) {
            echo "<div class='error'>$error</div>";
        }
    }
}
?>

Multiple Recipients (CC & BCC)

Send emails to multiple recipients with CC and BCC:

<?php
$mail = new PHPMailer(true);

// SMTP setup...

// Primary recipient
$mail->addAddress('primary@example.com', 'Primary Recipient');

// Additional recipients
$mail->addAddress('user1@example.com', 'User One');
$mail->addAddress('user2@example.com', 'User Two');

// CC (Carbon Copy) - visible to all
$mail->addCC('cc@example.com', 'CC Recipient');
$mail->addCC('another-cc@example.com');

// BCC (Blind Carbon Copy) - hidden from others
$mail->addBCC('bcc@example.com', 'BCC Recipient');

// Reply-To address
$mail->addReplyTo('reply@example.com', 'Reply Address');

$mail->Subject = 'Email to Multiple Recipients';
$mail->Body = 'This email is sent to multiple people';

$mail->send();
?>

Advanced Error Handling

Implement comprehensive error handling and debugging:

<?php
use PHPMailer\PHPMailer\PHPMailer;
use PHPMailer\PHPMailer\Exception;

$mail = new PHPMailer(true);

try {
    // Enable verbose debug output (for development only)
    $mail->SMTPDebug = 2; // 0=off, 1=client, 2=client+server, 3=detailed
    $mail->Debugoutput = 'html';
    
    // SMTP settings
    $mail->isSMTP();
    $mail->Host = 'smtp.gmail.com';
    $mail->SMTPAuth = true;
    $mail->Username = 'your-email@gmail.com';
    $mail->Password = 'your-app-password';
    $mail->SMTPSecure = PHPMailer::ENCRYPTION_STARTTLS;
    $mail->Port = 587;
    
    // Set timeout
    $mail->Timeout = 30; // seconds
    
    $mail->setFrom('from@example.com');
    $mail->addAddress('to@example.com');
    $mail->Subject = 'Test';
    $mail->Body = 'Test message';
    
    // Verify connection before sending
    if (!$mail->smtpConnect()) {
        throw new Exception('Could not connect to SMTP server');
    }
    
    $mail->send();
    echo 'Message sent successfully!';
    
} catch (Exception $e) {
    // Log error
    error_log("PHPMailer Error: " . $e->getMessage());
    
    // User-friendly message
    echo "Sorry, we couldn't send your message. Please try again later.";
    
    // Detailed error (for development)
    // echo "Error: {$mail->ErrorInfo}";
}
?>
